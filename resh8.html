<.button onclick="удалитьПробелы()">Удалить пробелы<./button>

<.script>

function удалитьПробелы() {

var текст = document.getElementById("текст").value;

var текстБезПробелов = текст.replace(/\s/g, '');

document.getElementById("текст").value = текстБезПробелов;

}

<./script>
