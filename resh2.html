В HTML-разметке создается поле ввода числа с уникальным идентификатором numberInput и кнопка с атрибутом onclick, который вызывает функцию checkEvenOdd() при нажатии.

В JavaScript-коде функция checkEvenOdd() получает значение введенного числа из поля ввода, преобразует его в целое число и проверяет его на четность.

В зависимости от результата проверки, программа выводит соответствующее сообщение о четности или нечетности числа. <.label for="numberInput">Введите число:

<.input type="number" id="numberInput">

<.button onclick="checkEvenOdd()">Проверить

<.p>id="result"<.p>>

function checkEvenOdd() { // Получаем значение введенного числа из поля ввода

var number = parseInt(document.getElementById("numberInput").value);

// Проверяем четность числа

if (number % 2 === 0) {

document.getElementById("result").innerText = "Число " + number + " четное.";

} else {

document.getElementById("result").innerText = "Число " + number + " нечетное."; } }
