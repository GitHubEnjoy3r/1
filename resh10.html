using System;

using System.Windows.Forms;

namespace DisplayArrayElements {

public partial class Form1 : Form {

private string[] массив = { "Элемент 1", "Элемент 2", "Элемент 3", "Элемент 4", "Элемент 5" }; private int currentIndex = 0;

public Form1() { InitializeComponent();

} private void button1_Click(object sender, EventArgs e)

{ if (currentIndex < массив.Length)

{ MessageBox.Show(массив[currentIndex]);

currentIndex++; }

else {

MessageBox.Show("Все элементы массива отображены"); currentIndex = 0; // Сброс индекса для повторного отображения массива

}

}

}
