<.button id="sendRequestBtn">Отправить запрос на сервер 1<./button>

<.script>

document.getElementById('sendRequestBtn').addEventListener('click', function() {

fetch('/sendRequest1', {

method: 'POST',

body: JSON.stringify({ requestType: 'my1Response' }),

headers: {

'Content-Type': 'application/json'

}

})

.then(response => {

if (response.ok) {

console.log('Запрос 1 принят');

} else {

console.error('Ошибка при выполнении запроса');

}

})

.catch(error => {

console.error('Ошибка:', error);

});

});

<./script>

server.js

const express = require('express');

const bodyParser = require('body-parser');

const app = express();

const port = 3000;

app.use(bodyParser.json());

app.post('/sendRequest1', (req, res) => {

const requestData = req.body;

if (requestData.requestType === 'my1Response') {

console.log('Запрос 1 принят');

res.status(200).send('Запрос 1 принят');

} else {

console.error('Ошибка: Неверный тип запроса');

res.status(400).send('Ошибка: Неверный тип запроса');

}

});

app.use(express.static('public'));

app.listen(port, () => {

console.log(`Сервер запущен на порту ${port}`);

});
